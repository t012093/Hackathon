インストール手順

# VSCodeマーケットプレイスから
code --install-extension saoudrizwan.cline

# 必要な権限設定
- ワークスペースアクセス
- ファイル操作権限
- ターミナル実行権限
⚙️

設定ファイル

{
  "cline.workspaceDirectory": "/path/to/project",
  "cline.autoApprove": true,
  "cline.theme": "light"
}

🔌 MCPツール例

// Brave検索の使用
use_mcp_tool {
  server_name: "brave-search",
  tool_name: "brave_web_search",
  arguments: {
    query: "最新のReactパターン",
    count: 5
  }
}

// Puppeteerでのスクリーンショット
use_mcp_tool {
  server_name: "puppeteer",
  tool_name: "puppeteer_screenshot",
  arguments: {
    name: "homepage",
    selector: ".main-content"
  }
}
