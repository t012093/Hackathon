自動化の活用

📍 自動化ルール例

# 自動化設定例
name: Project Board Automation

on:
  issues:
    types: [opened, closed]
  pull_request:
    types: [opened, closed]

jobs:
  automate_projects:
    runs-on: ubuntu-latest
    steps:
      - name: Move to column
        uses: alex-page/github-project-automation-plus@v0.8.3
        with:
          project: "Project Name"
          column: ${{ github.event_name == 'opened' ? 'To do' : 'Done' }}
